#####   5. adopt_long (Take control of EXIT by Brain)
It is a recovery function for a rare situation where a real deal exists but BrainFVVO is no longer tracking it and want Brain take control of EXIT


$manualSecret = "BrainFVVO_DEMO_MANUAL_3h2l9a7m6i3s8q7z9k1ll"
$manualUrl = "https://braindemolong-production.up.railway.app/manual"
$actual3CommasEntryPrice = 68.96
$body = @{
  secret = $manualSecret
  action = "adopt_long"
  symbol = "BINANCE:SOLUSDT"
  entryPrice = $actual3CommasEntryPrice
  profile = "CROSS_UP_CONFIRM"
  reason = "demo_adopt_existing_3commas_deal"
} | ConvertTo-Json
Invoke-RestMethod `
  -Method Post `
  -Uri $manualUrl `
  -ContentType "application/json" `
  -Body $body
