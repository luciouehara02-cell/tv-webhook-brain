1. Manual entry — start deal now (START)

$body = @{
  secret = "BrainFVVO_MANUAL_3h2l9a7m6i3s8q7z9k1ll"
  action = "enter_long"
  symbol = "BINANCE:SOLUSDT"
  profile = "FEATURE_CROSS_CONTINUATION"
  reason = "manual_entry_latest_price"
} | ConvertTo-Json
Invoke-RestMethod `
  -Method Post `
  -Uri "https://braindemolong-production.up.railway.app/manual" `
  -ContentType "application/json" `
  -Body $body
